pipeline {
    agent any

    stages {
        stage('1. Préparation') {
            steps {
                echo 'Nettoyage du workspace...'
                cleanWs()
            }
        }

        stage('2. Analyse SonarQube') {
            steps {
                // Remplacez 'MonSonarLocal' par le nom exact donné dans Jenkins
                withSonarQubeEnv('MonSonarLocal') {
                    echo 'Simulating SonarQube Scan...'
                    // Pour un vrai projet Node, on installerait le sonar-scanner
                }
            }
        }

        stage('3. Build de l\'image Docker') {
            steps {
                echo 'Construction de l\'image...'
                sh 'docker build -t mon-app-test:latest .'
            }
        }

        stage('4. Déploiement Local') {
            steps {
                echo 'Lancement du conteneur...'
                // Supprime l'ancien conteneur de test s'il existe, puis lance le nouveau
                sh 'docker stop app-test-instance || true && docker rm app-test-instance || true'
                sh 'docker run -d --name app-test-instance -p 8085:8080 mon-app-test:latest'
            }
        }
    }
}
